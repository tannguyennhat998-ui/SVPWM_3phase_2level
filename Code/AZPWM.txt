__interrupt void cpu_timer0_isr(void)
{
    CpuTimer0.InterruptCount++;

    t += 1.0;
    if (t > 199)
    {
        t = 0;
        EPwm1Regs.TBCTL.bit.SWFSYNC = 1;   // software sync pulse
        EPwm1Regs.TBCTL.bit.SWFSYNC = 0;
    }

    PieCtrlRegs.PIEACK.all = PIEACK_GROUP1;
}



void PWM()
{

    theta=goc-90;
    if(theta>180){theta=theta-360;}
    theta=theta*pi/180;

    if((theta>0)&(theta<=pi/3)) {n=1;}
    if((theta>pi/3)&(theta<=2*pi/3)) {n=2;}
    if((theta>2*pi/3)&(theta<=pi)) {n=3;}
    if((theta>-pi/3)&(theta<=0)) {n=6;}
    if((theta>-2*pi/3)&(theta<=-pi/3)) {n=5;}
    if((theta>-pi)&(theta<=-2*pi/3)) {n=4;}


    if(n==1)   {phi=theta;}
    if(n==2)   {phi=theta-pi/3;}
    if(n==3)   {phi=theta-2*pi/3;}
    if(n==4)   {phi=theta+pi;}
    if(n==5)   {phi=theta+2*pi/3;}
    if(n==6)   {phi=theta+pi/3;}



    t1=m*sin((pi/3.0)-phi);
    t2=m*sin(phi);
    t3=t6=(1-t1-t2)/2.0;

    if(n==1)
    {
    vau=1;
    vad=t3;
    vbu=t2+t3;
    vbd=0;
    vcu=1;
    vcd=t1+t2+t3;
    }

    if(n==2)
    {
    vau=1;
    vad=t2+t3;
    vbu=1-t6;
    vbd=0;
    vcu=t3;
    vcd=0;
    }

    if(n==3)
    {
    vau=1;
    vad=1-t6;
    vbu=1;
    vbd=t3;
    vcu=t2+t3;
    vcd=0;
    }

    if(n==4)
    {
    vau=t3;
    vad=0;
    vbu=1;
    vbd=t2+t3;
    vcu=1-t6;
    vcd=0;
    }

    if(n==5)
    {
    vau=t2+t3;
    vad=0;
    vbu=1;
    vbd=1-t6;
    vcu=1;
    vcd=t3;
    }

    if(n==6)
    {
    vau=1-t6;
    vad=0;
    vbu=t3;
    vbd=0;
    vcu=1;
    vcd=t2+t3;
    }
                     vap=vau;
                     van=vad;
                     vbp=vbu;
                     vbn=vbd;
                     vcp=vcu;
                     vcn=vcd;




}

void Setup(void)
{
    DINT;
    InitPieCtrl();
    IER = 0x0000;
    IFR = 0x0000;
    Interrupt_initModule(); // interrupt
    InitPieVectTable();
    EALLOW;
    PieVectTable.ADCA1_INT = &adc_isr;
    //Interrupt_register(INT_ADCA1, &adc_isr);
    EDIS;

    // Re-map Timer0 ISR vector
    EALLOW;
    PieVectTable.TIMER0_INT = &cpu_timer0_isr;
    EDIS;

    InitCpuTimers();
    ConfigCpuTimer(&CpuTimer0, 100, 100);     // 100 us period ,
    CpuTimer0Regs.TCR.bit.TIE = 1;               // enable Timer0 interrupt

    // Enable PIE group 1, INTx7 (Timer0)
    Interrupt_enable(INT_ADCA1); // Turn on ADC interrupt
    PieCtrlRegs.PIEIER1.bit.INTx7 = 1;
    IER |= 1;


    EINT;   // Global interrupt enable
    ERTM;   // Real-time debug

    CpuTimer0Regs.TCR.bit.TSS = 0; // start Timer0
}
void Init_ADC(void)
{
    EALLOW;
    AdcaRegs.ADCCTL2.bit.PRESCALE = 6;  // ADCCLK = SYSCLK / 4
    EDIS;

    ADC_setMode(ADCA_BASE, ADC_RESOLUTION_12BIT, ADC_MODE_SINGLE_ENDED);

    EALLOW;
    AdcaRegs.ADCCTL1.bit.ADCPWDNZ = 1;   // Turn on ADC
    AdcaRegs.ADCCTL1.bit.INTPULSEPOS = 1; // Interrupt when conversion is ended (mode)
    EDIS;

    DELAY_US(1000); // wait for ADC stabilization


    EALLOW;
    // SOC0
    AdcaRegs.ADCSOC0CTL.bit.ACQPS = 14;   // Sample time (t_sample = (ACQPS+1) * t_ADC_clock)
    AdcaRegs.ADCSOC0CTL.bit.CHSEL = 14;    // ADCINA14
    AdcaRegs.ADCSOC0CTL.bit.TRIGSEL = 0x05;  // Activate by EPWM1 SOCA
    EDIS;


    EALLOW;
    // SOC1
    AdcaRegs.ADCSOC1CTL.bit.ACQPS = 14;   // Sample time (t_sample = (ACQPS+1) * t_ADC_clock)
    AdcaRegs.ADCSOC1CTL.bit.CHSEL = 15;    // ADCINA14
    AdcaRegs.ADCSOC1CTL.bit.TRIGSEL = 0x05;  // Activate by EPWM1 SOCA
    EDIS;

    EALLOW;
    // ADC
    AdcaRegs.ADCINTSEL1N2.bit.INT1SEL = 1; // ADCINT1 interrupt for EOC
    AdcaRegs.ADCINTSEL1N2.bit.INT1E = 1;   // Set interrupt flag ADCINT1
    AdcaRegs.ADCINTFLGCLR.bit.ADCINT1 = 1; // Reset interrupt flag
    EDIS;


}
__interrupt void adc_isr(void)
{
    //Vin from 34-104
    // Read ADC value and store in buffer
    adc_val_c12 = AdcaResultRegs.ADCRESULT0; //0.8mV
    adc_val_c34 = AdcaResultRegs.ADCRESULT1; //0.8mV

    error=setpoint_vpn-(2*voltage_c34+2*voltage_c12);

    //PI=0.0002201*error-0.0002194*error_prev+PI_prev;
    /*
    if (error>15 || error<-15)
    {
    PI=0.006006*error-0.005994*error_prev+PI_prev;
    }
    else
    {
    PI=0.0002201*error-0.0002194*error_prev+PI_prev;
    }
    PI=0.006001*error-0.005999*error_prev+PI_prev;
    */
    PI=0.006001*error-0.005999*error_prev+PI_prev;
    error_prev=error;
    PI_prev=PI;

    // Reset interrupt flag
    AdcaRegs.ADCINTFLGCLR.bit.ADCINT1 = 1;
    PieCtrlRegs.PIEACK.all = PIEACK_GROUP1;
}
